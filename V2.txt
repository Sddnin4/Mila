-- Functions
local function loadScript(scriptData)
    -- Animation effect
    local button = scriptData.button
    if not button then return end
    
    -- Create loading animation
    local originalText = button.Text
    local originalColor = button.BackgroundColor3
    
    -- Loading state
    button.BackgroundColor3 = Color3.new(0.4, 0.8, 0.4)
    button.Text = "‚è≥ Loading..."
    
    -- Animate loading dots
    spawn(function()
        local dots = ""
        for i = 1, 3 do
            wait(0.2)
            dots = dots .. "."
            if button.Text:find("Loading") then
                button.Text = "‚è≥ Loading" .. dots
            end
        end
    end)
    
    -- Load script
    spawn(function()
        wait(0.8) -- Wait for animation
        pcall(function()
            local success, error = pcall(function()
                loadstring(game:HttpGet(scriptData.url))()
            end)
            
            if success then
                button.BackgroundColor3 = Color3.new(0.4, 0.9, 0.4)
                button.Text = "‚úÖ Loaded!"
            else
                button.BackgroundColor3 = Color3.new(0.9, 0.4, 0.4)
                button.Text = "‚ùå Error!"
                print("Script Error:", error)
            end
        end)
        
        wait(2)
        -- Reset button
        button.BackgroundColor3 = originalColor
        button.Text = originalText
    end)
end

local function createScriptButton(scriptData, position)
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0, 110, 0, 30)
    button.Position = position
    button.BackgroundColor3 = Color3.new(0.9, 0.95, 1)
    button.Text = "üóÉÔ∏è " .. scriptData.name
    button.TextColor3 = Color3.new(0.2, 0.3, 0.6)
    button.TextSize = 9
    button.Font = Enum.Font.Gotham
    button.TextWrapped = true
    button.Parent = scriptsFrame
    
    local buttonCorner = Instance.new("UICorner")
    buttonCorner.CornerRadius = UDim.new(0, 5)
    buttonCorner.Parent = button
    
    -- Store button reference in script data
    scriptData.button = button
    
    button.MouseButton1Click:Connect(function()
        loadScript(scriptData)
    end)
    
    -- Hover effects
    button.MouseEnter:Connect(function()
        if not button.Text:find("Loading") and not button.Text:find("Loaded") and not button.Text:find("Error") then
            button.BackgroundColor3 = Color3.new(0.8, 0.9, 1)
        end
    end)
    
    button.MouseLeave:Connect(function()
        if not button.Text:find("Loading") and not button.Text:find("Loaded") and not button.Text:find("Error") then
            button.BackgroundColor3 = Color3.new(0.9, 0.95, 1)
        end
    end)
    
    return button
end

local function displayScripts(filteredScripts)
    -- Clear existing buttons
    for _, child in pairs(scriptsFrame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end
    
    -- Create script buttons in grid (3 columns)
    local columns = 3
    local buttonWidth = 110
    local buttonHeight = 35
    local paddingX = 10
    local paddingY = 5
    
    for i, script in ipairs(filteredScripts) do
        local row = math.floor((i - 1) / columns)
        local col = (i - 1) % columns
        
        local posX = col * (buttonWidth + paddingX) + 5
        local posY = row * (buttonHeight + paddingY) + 5
        
        createScriptButton(script, UDim2.new(0, posX, 0, posY))
    end
    
   -- Update canvas size
    local totalRows = math.ceil(#filteredScripts / columns)
    scriptsFrame.CanvasSize = UDim2.new(0, 0, 0, totalRows * (buttonHeight + paddingY) + 10)
end

local function switchTab(tabName)
    currentTab = tabName
    
    -- Update tab buttons
    for name, tabData in pairs(tabButtons) do
        if name == tabName then
            tabData.button.TextColor3 = Color3.new(0.2, 0.4, 0.8)
            tabData.underline.Visible = true
        else
            tabData.button.TextColor3 = Color3.new(0.5, 0.5, 0.5)
            tabData.underline.Visible = false
        end
    end
    
    -- Update content visibility
    homeContent.Visible = tabName == "Home"
    scriptContent.Visible = tabName == "Script"
    notifContent.Visible = tabName == "Notifications"
    otherContent.Visible = tabName == "Other"
end

local function toggleNoclip()
    noclip = not noclip
    if noclip then
        local connection
        connection = RunService.Stepped:Connect(function()
            if not noclip then
                connection:Disconnect()
                return
            end
            for _, part in pairs(character:GetChildren()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end
        end)
    else
        for _, part in pairs(character:GetChildren()) do
            if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                part.CanCollide = true
            end
        end
    end
end

local function toggleInfiniteJump()
    infiniteJump = not infiniteJump
    if infiniteJump then
        jumpConnection = UserInputService.JumpRequest:Connect(function()
            if character and character:FindFirstChild("Humanoid") then
                character.Humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
            end
        end)
    else

            playerFrame.Visible = false

            mainFrame.Visible = true
            toggleButton.Visible = false
        end
    end
